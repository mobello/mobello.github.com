  30       /**
  31        * apply default value to destination object
  32        * @param {Object} dest destination object
  33        * @param {Object} defaultObj default object
  34        * @returns {Object} applied object
  35        */
  36       applyDefault: function(dest, defaultObj, deep) {
  37         var c, p;
  38         if (dest == null && !deep) {
  39           dest = defaultObj;
  40           return dest;
  41         }
  42         if ((typeof defaultObj) == "object") {
  43           c = defaultObj.constructor;
  44           switch (c) {
  45           case Object:
  46             dest = dest||{};
  47             break;
  48           case Array:
  49             dest = dest||[];
  50             break;
  51           default:
  52             dest = defaultObj;
  53             return dest;
  54           }
  55           for (p in defaultObj) {
  56             if (dest[p] === undefined) {
  57               if (deep) {
  58                 dest[p] = tau.util.applyDefault(undefined, defaultObj[p], deep);
  59               } else {
  60                 dest[p] = defaultObj[p];
  61               }
  62             }
  63           }
  64         } else {
  65           dest = defaultObj;
  66         }
  67         return dest;
  68       },

  70       /**
  71        * Get and set the style property on a HTMLElement
  72        * @param {HTMLElement} elem
  73        * @param {String} name
  74        * @param {String} value
  75        * @returns {String}
  76        */
  77       style: function(elem, name, value) {
  78         if (!tau.isElement(elem) || !name) return undefined;
  79 
  80         if ((name === "width" || name === "height") && parseFloat(value) < 0) {
  81           value = undefined;
  82         }
  83 
  84         var style = elem.style, 
  85             set = value !== undefined;
  86 
  87         name = tau.camelize(name);
  88         
  89         if (set) style[name] = value;
  90 
  91         return style[name];
  92       },
  93 
  94       /**
  95        * Get computed style on a HTMLElement
  96        * @param {HTMLElement} elem
  97        * @param {String} name
  98        * @param {Boolean} force
  99        * @returns {String}
 100        */
 101       getComputedStyle: function(elem, name, force) {
 102         var style = elem.style, val, currentStyle;
 103 
 104         if (!force && style && style[name]) {
 105           val = style[name];
 106         } else if (name === "width" || name === "height") {
 107           val = (name === "width") ? elem.clientWidth : elem.clientHeight;
 108           if (val == 0) {
 109             var prevStyle = {},
 110               props = {
 111                 position: "absolute",
 112                 visibility: "hidden",
 113                 display: "block"
 114               };
 115 
 116             for ( var key in props) {
 117               prevStyle[key] = style[key];
 118               style[key] = props[key];
 119             }
 120             val = (name === "width") ? elem.clientWidth : elem.clientHeight;
 121 
 122             for ( var key in props) {
 123               style[key] = prevStyle[key];
 124             }
 125           }
 126           val = Math.max(0, Math.round(val));
 127 
 128         } else if (document.defaultView && document.defaultView.getComputedStyle) {
 129           var defaultView = elem.ownerDocument.defaultView;
 130   
 131           if (!defaultView) return null; 
 132   
 133           var computedStyle = defaultView.getComputedStyle(elem, null);
 134 
 135           name = tau.underscore(name).replace(/_/g, '-');
 136 
 137           if (computedStyle) val = computedStyle.getPropertyValue(name); 
 138 
 139           if (name === "opacity" && val === "") val = "1"; 
 140 
 141         } else if (currentStyle = elem.currentStyle) {
 142           val = currentStyle[name] || currentStyle[tau.camelize(name)];
 143   
 144           if (/^-?(?:\d*\.)?\d+(?!px)[^\d\s]+$/i.test(val)) {
 145             style = elem.style;
 146             var left = style.left,
 147               runtimeStyle = elem.runtimeStyle,
 148               rsLeft = runtimeStyle.left;
 149             
 150             runtimeStyle.left = currentStyle.left;
 151             style.left = (name === "font-size") ? "1em" : (val || 0);
 152             val = style.pixelLeft + "px";
 153 
 154             style.left = left;
 155             runtimeStyle.left = rsLeft;
 156           }
 157         }
 158         return val;
 159       },

 164       setCookie: function (name, value, days) {
 165         var date, expires = '';
 166         if (days) {
 167           date = new Date();
 168           date.setTime(date.getTime() + (days * 24 * 3600 * 1000));
 169           expires = "; expires=" + date.toGMTString();
 170         }
 171         document.cookie = name.concat('=', encodeURI(value), expires, '; path=/');
 172       },

 177       getCookie: function (name) {
 178         var cookie,
 179             delimiter = name.concat('='),
 180             cookies = document.cookie.split(';');
 181         for (var i = cookies.length - 1; i >= 0; i--) {
 182           cookie = cookies[i].replace(/^\s+/, ''); // LTrim
 183           if (cookie.indexOf(delimiter) == 0) {
 184             return decodeURI(cookie.substring(delimiter.length));
 185           }
 186         }
 187         return null;
 188       },

 207     /**
 208      * Determines whether this dom element has the given className.
 209      * @param {Object} dom element
 210      * @param {String} className
 211      * @returns {Boolean}
 212      */
 213     hasClass: function(element, className, prefix) {
 214       if (element) {
 215         var classList = element.classList;
 216         className = (prefix || '') + className;
 217         if (classList){
 218           return classList.contains(className);
 219         } else {
 220           var elementClassName = element.className;
 221           return (elementClassName.length > 0 && (elementClassName == className || new RegExp(
 222               "(^|\\s)" + className + "(\\s|$)").test(elementClassName)));
 223         }
 224       }
 225       return false;
 226     },
 227 
 228     /**
 229      * Adds a class name to a given DOM element.
 230      * @param {Object} dom element
 231      * @param {String} className
 232      * @param {String} [prefix]
 233      * @returns {Boolean}
 234      */
 235     addClass: function(element, className, prefix) {
 236       // Tag the DOM node as an EventDelegator
 237       if (!element || !className) {
 238         return false;
 239       }
 240       className = (prefix || '') + className;
 241       
 242       var classList = element.classList;
 243       if (classList){
 244         classList.add(className);
 245         return true;
 246       } else {
 247         if (!tau.util.dom.hasClass(element, className)) {
 248           element.className += (element.className ? ' ' : '') + className;
 249           return true;
 250         }
 251       }
 252       return false;
 253     },
 254 
 255     /**
 256      * Removes a class name from a given element.
 257      * @param {Object} dom element
 258      * @param {String} className
 259      * @returns {Boolean}
 260      */
 261     removeClass: function(element, className, prefix) {
 262       if (element && className) {
 263         className = (prefix || '') + className;
 264         var classList = element.classList;
 265         if (classList){
 266           classList.remove(className);
 267         } else {
 268           element.className = element.className.replace(
 269               new RegExp("(^|\\s+)" + className + "(\\s+|$)"), ' ')
 270               .replace(/^\s+/, '').replace(/\s+$/, '');
 271         }
 272         return true;
 273       }
 274       return false;
 275     },

 277     /**
 278      * Replace a class with another class for a given element. If no
 279      * oldClassName is present, the newClassName is simply added.
 280      * @param {Object} dom element
 281      * @param {String} oldClassName the class name to be replaced
 282      * @param {String} newClassName the class name that will be replacing
 283      *        the old class name
 284      * @retruns {Boolean}
 285      */
 286     replaceClass: function(element, oldC, newC, prefix) {
 287       if (element){
 288         if (oldC) {
 289           oldC = (prefix || '') + oldC;
 290         }
 291         if (newC){
 292           newC = (prefix || '') + newC;
 293         }
 294         
 295         var classList = element.classList;
 296         if (classList){
 297           if (oldC){
 298             classList.remove(oldC);
 299           }
 300           classList.add(newC);
 301         } else {
 302           if (tau.util.dom.hasClass(element, oldC)) {
 303             element.className = element.className.replace(
 304                 new RegExp("(^|\\s+)" + oldC + "(\\s+|$)"), ' ' +newC).replace(
 305                     /^\s+/, '').replace(/\s+$/, '');
 306           } else {
 307             tau.util.dom.addClass(element, newC);
 308           }
 309         }
 310         return true;
 311       }
 312       return false;
 313     },

 327     /**
 328      * Wrapper for HTMLElement method.
 329      * @method appendChild
 330      * @param {tau.ui.Component || HTMLElement} parent
 331      * @param {tau.ui.Component || HTMLElement} child The element to
 332      *        append.
 333      * @return {HTMLElement} The appended DOM element.
 334      */
 335     appendChild: function(parent, child) {
 336 
 337       var parentElem = null;
 338       var childElem = null;
 339 
 340       if (tau.isElement(parent) || tau.isFragment(parent)) {
 341         parentElem = parent;
 342       } else if (parent instanceof tau.ui.Component) {
 343         parentElem = tau.util.dom.elementOf(parent.getId());
 344       }
 345       if (!parentElem)
 346         return;
 347 
 348       if (tau.isElement(child) || tau.isFragment(child)) {
 349         childElem = child;
 350       } else if (child instanceof tau.ui.Component) {
 351         childElem = tau.util.dom.elementOf(child.getId());
 352       }
 353       if (!childElem)
 354         return;
 355       parentElem.appendChild(childElem);
 356     },

 358     /** 
 359      * Removes all DOM elements from their parent (bottom-up).
 360      * @param {Object|Array} elem DOM element, can pass more than one parameters
 361      * @returns {Object|Array} Array or removed elements, or a single element 
 362      */
 363     removeElements: function (elem) {
 364       var i, e, r, elems, removed; 
 365       if (arguments.length === 1 && elem.parentNode) {
 366         return elem.parentNode.removeChild(elem); // Remove a single DOM element
 367       } else if (arguments.length > 1) {
 368         elems = arguments;                        // Multiple parameters
 369       } else if (!tau.isString(elem) && elem.length) {
 370         elems = elem;                             // Single array parameter
 371       }
 372 
 373       // Handles Paramter/Element list removal
 374       removed = [];
 375       if (elems) {
 376         for (i = elems.length - 1; i >= 0; i--) {
 377           e = elems[i];
 378           r = null;
 379           if (e.length) {
 380             r = tau.util.dom.removeElements(e); // Recursively remove a list
 381           } else if (e.parentNode) {
 382             r = e.parentNode.removeChild(e);    // Remove single DOM directly
 383           }
 384 
 385           // Update removed DOM list
 386           if (tau.isArray(r)) {
 387             removed.unshift.apply(removed, r);
 388           } else if (r) {
 389             removed.unshift(r);
 390           }
 391         }
 392       }
 393       return removed;
 394     },

 396     /**
 397      * Pushes a DOM element to the top of a parent node.
 398      * @param {Object} parent DOM element to push the node
 399      * @param {Object} elem DOM element to be pushed
 400      * @param {Object} handleDisplay True will show pushed DOM and hide old
 401      * @returns {Boolean} True if the element was pushed properly
 402      */
 403     pushElement: function (parent, elem, handleDisplay) {
 404       if (handleDisplay && elem) { 
 405         elem.style.display = '';
 406       }
 407       if (parent.children && parent.children[0]) {
 408         if (handleDisplay) {
 409           parent.children[0].style.display = 'none';
 410         }
 411         return parent.insertBefore(elem, parent.children[0]) !== null;
 412       } else {
 413         return parent.appendChild(elem) !== null;
 414       }
 415     },

 417     /**
 418      * Pops a DOM element from the top of a parent node.
 419      * @param {Object} parent DOM element to pop the element from
 420      * @param {Object} elem DOM element to pop
 421      * @param {Object} handleDisplay True will show pushed DOM and hide old
 422      */
 423     popElement: function (parent, elem, handleDisplay) {
 424       if (!parent) {
 425         parent = elem.parent;
 426       }
 427       if (parent && elem && parent.removeChild(elem)) {
 428         if (handleDisplay && parent.children && parent.children[0]) {
 429           parent.children[0].style.display = '';
 430         }
 431         return elem;
 432       }
 433       return null;
 434     },

 436     /**
 437      * Returns the style value from CSSRules.
 438      * @method getStyleFromCssRule
 439      * @param {String} selector
 440      * @param {String} prop
 441      * @return {String} style value
 442      */
 443     getStyleFromCssRule : function(selector, prop) {
 444       prop = tau.camelize(prop);
 445       var oStyleSheet = document.styleSheets;
 446       if (oStyleSheet){
 447         for(var i=0; i < oStyleSheet.length; i++){
 448           var cssRules = oStyleSheet[i].cssRules || oStyleSheet[i].rules;
 449           if (cssRules){
 450             for(var j=0; j < cssRules.length; j++){
 451               if (cssRules[j].selectorText === selector){
 452                 return cssRules[j].style[prop];
 453               }
 454             }
 455           }
 456         }
 457       }
 458       return null;
 459     },

 464      * tau.util.dom.insertStyleRule("#componentId {height : 200px !important; width : 200px !important;}");
 465      * @param {String} rule select style declaration .
 466      * @return {Number} index  .   -1 .
 467      */
 468     insertStyleRule: function (rule) {
 469       var styleSheets = document.styleSheets,
 470           styleSheet = styleSheets[styleSheets.length - 1],
 471           index = styleSheet.cssRules ? styleSheet.cssRules.length - 1 : 0;
 472       if (index < 0){
 473         index = 0;
 474       }
 475       if (styleSheet){
 476         styleSheet.insertRule(rule, index);
 477         return index;
 478       }
 479       return -1;
 480     },

 497     /**
 498      * Gets the current position of the element based on page coordinates.  
 499      * Element must be part of the DOM tree to have page coordinates 
 500      * (display:none or elements not appended return false).
 501      * @param {Object} DOM element
 502      * @return {Array} The XY position of the element
 503      */
 504      getXY : function(elem) {
 505          var body = document.getElementById('tau-root'),
 506              dom = parent = elem,
 507              x = y = 0;
 508   
 509          if (!dom || dom === body) {
 510              return [0, 0];
 511          }
 512   
 513          while (parent) {
 514              x += parent.offsetLeft;
 515              y += parent.offsetTop;
 516   
 517              if(parent != dom) {
 518                  // For webkit, we need to add parent's clientLeft/Top as well.
 519                  x += parent.clientLeft || 0;
 520                  y += parent.clientTop || 0;
 521              }
 522   
 523              parent = parent.offsetParent;
 524          }
 525   
 526          // Safari absolute incorrectly account for body offsetTop.
 527          if (tau.rt.isWebkit && tau.util.style(elem, 'position') === 'absolute') {
 528              y -= body.offsetTop;
 529          }
 530   
 531          parent = dom.parentElement;
 532          while (parent && parent != body) {
 533              x -= parent.scrollLeft;
 534              y -= parent.scrollTop;
 535              parent = parent.parentElement;
 536          }
 537          return [x, y];
 538      },
