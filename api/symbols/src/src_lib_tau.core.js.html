 423       /**
 424        * Augments class properties to an object instance/class.
 425        * <p/>
 426        * Augmenting an object instance will give properties only to that 
 427        * instance. Augmenting an object's prototype will give the properties 
 428        * to all instances derived by from that object via the new operator.
 429        * <p/>
 430        * Additionally, the augmented class' constructor will be called via 
 431        * proxy upon any one of its function call.
 432        * @param {Object} dest Destination object/class to apply the properties
 433        * @param {Object} src Source class to receive the properties from
 434        * @param {Boolean} override Override existing properties to destination
 435        * @param {Array} filter List of property names to filter from src
 436        * @returns {Object} Destination object/class with the applied properties
 437        * @throws {Error} Destination/source parameters must not be invalid
 438        */
 439       augment: function (dest, src, override, filter) {
 440         var p, srcProp, makeProxyFn, proxyFn, originalFn,
 441             isDestProto = tau.isObject(dest.prototype),
 442             destProto = dest.prototype,
 443             destTarget = dest.prototype || dest,
 444             srcProto = src.prototype;
 445 
 446         // Prototype augmentation: make proxy functions to call constructor
 447         if (isDestProto && src) {
 448           proxyFn = {};
 449           originalFn = {};
 450           /** @inner Maps an original function to a proxy function. */
 451           makeProxyFn = function (m) {
 452             /**
 453              * Proxy members functions are assigned to each original members. 
 454              * When one is called, it will execute the instance's contructor, 
 455              * so that it can inherit any of its this assigments, and re-attach
 456              * all the original members and properties.
 457              * @inner
 458              */
 459             proxyFn[m] = function () {
 460               // Re-attach all the original properties prior to constructor call
 461               for (var i in originalFn) {
 462                 if (originalFn.hasOwnProperty(i) && (this[i] === proxyFn[i])) {
 463                   this[i] = originalFn[i];
 464                 }
 465               }
 466               src.apply(this); // src constructor call to inherit this props
 467               return originalFn[m].apply(this, arguments); // original function
 468             };
 469             return proxyFn[m];
 470           };
 471 
 472           // Iterate through source properties and create/attach proxy props
 473           for (p in srcProto) {
 474             if (srcProto.hasOwnProperty(p)) {
 475               if ((!tau.isArray(filter) || (filter.indexOf(p) > -1))
 476                   && (override || !destProto[p])) {
 477                 srcProp = srcProto[p];
 478                 if (tau.isFunction(srcProp)) {
 479                   // Save original properties to originalFn and attach the proxy
 480                   originalFn[p] = srcProp;
 481                   destProto[p] = makeProxyFn(p);
 482                 } else {
 483                   // No need to create proxy members, just attach the property
 484                   destProto[p] = srcProp;
 485                 }
 486               }
 487             }
 488           }
 489         }
 490 
 491         // Mixes source prototype properties to destTarget (prototype)
 492         tau.mixin(destTarget, srcProto, override, filter);
 493 
 494         // Object instance augmentation: call source constructor to get props
 495         if (!isDestProto) {
 496           src.apply(destTarget);
 497         }
 498 
 499         return destTarget;
 500       },

 502       /**
 503        * Declares a namespace.
 504        * <p/>
 505        * An undefined namespace and any of its sub parts will be declared 
 506        * without overwriting any previously defined namespaces hierarchy.
 507        * <p/>
 508        * An exception occurs if the namespace syntax is incorrect or a function
 509        * attempts to overwrite an existing namespace. 
 510        * @param {String} ns Namespace declaration
 511        * @param {Object|Function} attach Properties/Function to add to namespace
 512        * @param {Boolean} overwrite Overwrites existing namespace
 513        * @returns {Boolean} true if at least one namespace part was modified
 514        * @throws {Error} Invalid namespace format or existing namespace
 515        */
 516       namespace: function (ns, attach, overwrite) {
 517         var i, len, parts, isFn, parent,
 518             modified = false;
 519         if (!tau.isNamespace(ns)) {
 520           throw new SyntaxError('tau.namespace: invalid namespace, ' + ns);
 521         }
 522 
 523         parent = GLOBAL;
 524         parts = ns.split('.');
 525         isFn = tau.isFunction(attach);
 526         // Validate each namespace parts (create one if it doesn't exist)
 527         for (i = 0, len = isFn ? parts.length - 1 : parts.length; i < len; i++) {
 528           if (typeof parent[parts[i]] == 'undefined') {
 529             parent[parts[i]] = {};
 530             modified = true;
 531           } else if (!overwrite && tau.isFunction(parent[parts[i]]) 
 532               && i < len - 1) {
 533             // Any parent part in the namespace must not be a function
 534             throw new Error('tau.namespace: \"' + parts.slice(0, i + 1)
 535                 .join('.') + '\" function was previously defined in hierarchy');
 536           }
 537           parent = parent[parts[i]];
 538         }
 539 
 540         if (isFn) {
 541           // Function assignment for the last portion of namespace
 542           if (overwrite || typeof parent[parts[i]] == 'undefined') {
 543             parent[parts[i]] = attach;
 544             modified = true;
 545           } else {
 546             throw new Error('tau.namespace: \"' + ns + '\" definition ' 
 547                 + ' already exists, cannot be overridden');
 548           }
 549         } else {
 550           // Attach properties to the final namespace part; does not overwrite
 551           for (i in attach) {
 552             if (typeof parent[i] == 'undefined' && attach.hasOwnProperty(i)) {
 553               parent[i] = attach[i];
 554               modified = true;
 555             }
 556           }
 557         }
 558 
 559         return modified;
 560       },

 563        * Clones a new instance of an object, function, array, dates & RegExp.
 564        * <pre>
 565        *   var original = new tau.ScriptHelper();
 566        *   alert(original !== tau.clone(original)); // true
 567        * </pre>
 568        * @param {Object} obj Object to clone
 569        * @param {Boolean} deep Recursively deep copy inner property instances
 570        * @param {Array} ignore List of properties to not deep clone
 571        * @returns {Object} New cloned version of the parameter
 572        */
 573       clone: function (obj, deep, ignore) {
 574         var F, cloned, p, 
 575             t = (obj instanceof HTMLElement) ? null : tau.typeOf(obj);
 576         switch (t) {
 577         case 'object':
 578         case 'error':
 579           if (obj.constructor === Object.prototype.constructor) {
 580             cloned = {}; // JSON object
 581           } else {
 582             /** @inner Cloning instance's state is preserved as a new Class. */
 583             F = function () {};
 584             F.prototype = obj; // obj changes are applied to cloned's prototype
 585             cloned = new F();
 586           }
 587           break;
 588         case 'function':
 589           /** @inner Cloning function is callbed by a new function. */
 590           cloned = function () {
 591             obj.apply(this, arguments);
 592           };
 593           cloned.prototype = obj.prototype;
 594           break;
 595         case 'array':
 596           cloned = [];
 597           break;
 598         case 'date':
 599           cloned = new Date(obj);
 600           break;
 601         case 'regexp':
 602           cloned = new RegExp(obj.source);
 603           break;
 604         default:
 605           return obj;
 606         }
 607 
 608         // Copy, or recursively deep copy, properties to cloned's 'this'
 609         for (p in obj) {
 610           if (obj.hasOwnProperty(p) 
 611               && (!ignore || ignore.indexOf(p) === -1)) {
 612             cloned[p] = !deep ? obj[p] : tau.clone(obj[p], deep, ignore);
 613           }
 614         }

 620        * <p/>

 622        * parameter; any parent properties with the identical name in the class
 623        * hierarchy will be overridden in the child class.
 624        * <p/>
 625        * Additionally, extended class will have a <code>superclass</code> 
 626        * property to access its parent methods and properties.
 627        * <p/>
 628        * Note that the child class must call the parent constructor in its own
 629        * constructor to property inherit declaration made by the parent.
 630        * @example
 631        * function tau.example.MySubClass() {
 632        *   tau.example.MySubClass.superclass.constructor.apply(this, arguments);
 633        * }
 634        * tau.extend(tau.example.MySubClass, tau.example.MyClass);
 635        * @param {Object} child Child/Sub class to apply properties to
 636        * @param {Object} parent Parent/Super class to inherit properties from
 637        * @param {Object} props Properties to add/override to the child class
 638        * @returns {Object} New extended class with the applied properties
 639        * @throws {Error} Parent/child parameter must be classes (functions).
 640        */
 641       extend: function (child, parent, props) {
 642         var p, F;
 643         // Only allow Classes (functions) as parameters
 644         if (!tau.isFunction(child) || !tau.isFunction(parent)) {
 645           throw new Error('tau.extend: parameters must be classes.');
 646         }
 647 
 648         // Create prototype inheritance references for the child class
 649         /** @inner */
 650         F = function () {};
 651         F.prototype = parent.prototype;
 652         child.prototype = new F();
 653         child.prototype.constructor = child;
 654 
 655         // Child class can call parent's methods via the superclass property.
 656         // If the parent's constructor is the Object itself, we must reassign
 657         // the constructor to itself for proper super call
 658         child.superclass = parent.prototype;
 659         if (parent != Object
 660             && parent.prototype.constructor == Object.prototype.constructor) {
 661           parent.prototype.constructor = parent;
 662         }
 663 
 664         // Add/Override extended functions the child class
 665         for (p in props) {
 666           if (props.hasOwnProperty(p)) {
 667             child.prototype[p] = props[p];
 668           }
 669         }
 670
